from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt
import numpy as np


thL = [1,2,3] #degree
for i in range(len(thL)):

    g = 9.8  # (m/s^2)
    L = 1.0  # (m)

    # 
    def pend1(t, state):
        dydt = np.zeros_like(state)
        #dydt0
        #\theta
        
        dydt[0] = state[1] # d/dt =  
        
        dydt[1] = -(g/L)*np.sin(state[0]) # d/dt = -(g/L)sin() 
        
        #print(dydt[0])
        return dydt

    #
    t_span = [0,20] # (s)
    dt = 0.05#  (s)
    t = np.arange(t_span[0], t_span[1], dt)# array 020dt

    # th1
    th1 = thL[i]#  [deg]
    w1 = 0.0 #  [deg/s]
    state = np.radians([th1, w1])# 

    #
    sol = solve_ivp(pend1, t_span, state, t_eval=t)

    theta = sol.y[0] #y[0]y[1]=
    omega = sol.y[1] #y[0]y[1]=
    #sol(t)(t)(t)

    plt.plot(t, theta, '-', label="theta = "+str(th1)+" "+"deg")
    plt.legend(loc="upper right")
    plt.xlabel("Time, $t$")
    plt.ylabel("Angle, theta")
